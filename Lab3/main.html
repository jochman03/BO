import numpy as np

def DPA(G, a, s):
    suma = 0 # sumaryczna waga krawdzi
    Q = [] # zbir wierzchokw nienalecych do MST
    A = [] # zbir krawdzi MST  przechowuje krawedzie jako krotki (wierzchoek, wierzchoek)
    alfa = {} # sownik przechowujcy dane   wierzchoek -> jego poprzednik
    beta = {} # sownik przechowujcy dane   wierzchoek -> waga krawdzi z MST
    for u, neighbours in G.items():    # dla kadego wierzchoka z grafu
        alfa[u] = 0    # jego poprzedni wierzchoek to 0 (inicjalizacja)
        beta[u] = np.inf # waga krawdzi do MST to nieskoczono (inicjalizacja)
        Q.append(u)  # Dodawanie wierzchoka do zbioru wierzchokw nienalecych do MST
    beta[s] = 0    # Odlego pocztkowego wierzchoka od MST to 0
    Q.remove(s)   # Usuwanie pocztkowego wierzchoka z Q
    u_last = s    # Ostatnim sprawdzanym wierzchokiem by pocztkowy
    while len(Q) != 0:  # Gwna ptla dla wszystkich niesprawdzonych
        for u in Q:   # dla kadego wierzchoka w Q
            if u in G[u_last]:  # Jeli ssiaduje z ostatnio sprawdzanym wierzchokiem
                if a(u, u_last) < beta[u]:   # jeli waga ich krawdzi jest mniejsza ni wierzchoka z MST
                    alfa[u] = u_last  # Poprzednikiem tego wierzchoka by ostatnio sprawdzany
                    beta[u] = a(u, u_last)  # Odlego tego wierzchoka od MST to waga krawdzi aktualnie sprawdzanego z poprzednio
        arg_min = None # deklaracja arg_min potrzebna do wyszukania wierzchoka z najmniejsz bet
        min_beta = np.inf # skoro najmniejsza na pocztku jest inf to kada != inf bdzie mniejsza
        for u in Q:  
            if beta[u] < min_beta: # jeli beta aktualnego wierzchoka jest mniejsza od ostatnio sprawdzanej
                min_beta = beta[u]   # to aktualny ma mniejsz
                arg_min = u # zostanie wyznaczony jako potencjalny najmniejszy
        if arg_min == None: # Sprawdzenie czy arg_min == None, jeli tak, to graf niespjny
            break  # przerywa poszukiwania
        u_last = arg_min  # znaleziono wierzchoek o najmniejszym beta
        Q.remove(u_last)  # usuwanie go z Q
        A.append((alfa[u_last], u_last))  # Dodawanie krotki reprezentujcej krawdz do MST
        suma = suma + a(alfa[u_last], u_last)  # Dodanie wagi krawdzi do caociowej sumy
    return (A, suma)

def print_graph_mst(graph1, edges2):
    G1 = nx.Graph()
    for node, neighbors in graph1.items():
        for neighbor in neighbors:
            G1.add_edge(node, neighbor)
    
    G2 = nx.Graph()
    G2.add_edges_from(edges2)
    
    pos = nx.spring_layout(G1)
    
    plt.figure(figsize=(6, 6))
    
    nx.draw(G1, pos, with_labels=True, node_color='lightblue', edge_color='gray', node_size=500, font_size=14, alpha=0.6)
    nx.draw(G2, pos, with_labels=True, node_color='lightblue', edge_color='red', node_size=500, font_size=14, alpha=0.6)
    
    plt.show()

Zadanie 3

Algorytm Kruskala polega na wyznaczeniu minimalnego drzewa rozpinjcego poprzez posortowanie krawdzi grafu wzgldem wag rosnco. Nastpnie do zbioru minimalnego drzewa rozpinajcego dodawane s wierzchoki, kte nie powtrzyy si ju wczeniej, w ten sposb unika si cyklu. Najwiksz zoono w algorytmie Kruskala ma operacja sortowania krawdzi po wagach. Zoono sortowania krawdzi wynosi O(ElogE), przy czym E to ilo krawdzi, przy czym pozostae operacje realizowane s ze zoonoci O(E * a(E, V)), gdzie V to liczba krawdzi grafu. Funkcja a ronie stosunkowo wolno

Zadanie 4

Podstawowym wykorzystaniem problemu MST jest projektowanie wszelkiego typu sieci, w ktrych naley doprowadzi poczenie do kadego odbiorcy po jak najmniejszej drodze. Przykadem moe by sie elektryczna. Po dokonaniu odpowiednich uproszcze problem mona zredukowa do znalezienia minimalnego drzewa rozpinajcego dla grafu, w ktrym wierzchokiem pocztkowym jest elektrownia, a kolejnymi wierzchokami odbiorcy (bardzo uproszczony model). Wagami mog by koszty transportu energi, odlegoci oraz problemy z infrastruktur. Aby model jak najlepiej odzwierciedla rzeczywiste warunki problemu naley w kosztach wag uwzgldni problemy poczeniowe. Poczenia zwykle nie mog by realizowane wprost od jednego odbiorcy do drugiego, naley uwzgldni obcienie sieci oraz potencjalne stacje transformatorowe. Algorytm mgby by zastosowany na dwa sposoby. Najpierw elektrownia - stacje transformatorowe z uwzgldnieniem infrastruktury i potencjalnych problemw. Nastpnie mgby by wykorzystany drugi raz od stacji transformatorowej do odbiorcw w najbliszej okolicy.
